151bool sendPacket(int socket, uint8_t type, uint32_t id, double timestamp = 0.0, double fuel_remaining = 0.0) {
152 AircraftPacket pkt;
153 pkt.message_type = type;
154 pkt.plane_id = id;
155
156 // Set data fields. For 0x01/0x03, these are the "Null Values"
157 pkt.timestamp = timestamp;
158 pkt.fuel_remaining = fuel_remaining;
159
160 size_t bytesToSend = (type == 0x02) ? 21 : 5;
161
162 if (send(socket, (char*)&pkt, bytesToSend, 0) == -1) {
163 return false;
164 }
165 return true;
166}

190bool processTelemetryFile(const std::string& filename, int socket, uint32_t planeId) {
191 std::ifstream file(filename);
192
193 if (!file.is_open()) {
194 std::cerr << "Error: Could not open telemetry file: " << filename << std::endl;
195 return false;
196 }
197
198 std::string line;
199 int failedPacketCounter = 0;
200
201 while (std::getline(file, line)) {
202 if (line.empty()) continue;
203
204 std::stringstream ss(line);
205 std::string ts_str, fuel_str;
206
207 if (std::getline(ss, ts_str, ',') && std::getline(ss, fuel_str, ',')) {
208 try {
209 double timestamp = std::stod(ts_str);
210 double fuelRemaining = std::stod(fuel_str);
211
212 if (!sendPacket(socket, 0x02, planeId, timestamp, fuelRemaining))
213 failedPacketCounter++;
214
215 if (failedPacketCounter == 5)
216 return false;
217 // 0.75s sleep
218 std::this_thread::sleep_for(std::chrono::milliseconds(750));
219 }
220 catch (const std::exception& e) {
221 // Log bad data but continue processing
222 std::cerr << "Skipping malformed line: " << line << " (" << e.what() << ")" << std::endl;
223 }
224 }
225 }
226
227 file.close();
228 return true;
229}
