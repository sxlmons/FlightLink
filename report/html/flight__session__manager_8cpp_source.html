60FlightSession start_session(uint32_t plane_id) {
61 FlightSession session{};
62 session.plane_id = plane_id;
63 session.prev_fuel = 0.0;
64 session.fuel_sum = 0.0;
65 session.packet_count = 0;
66 session.has_baseline = false;
67
68 {
69 std::lock_guard<std::mutex> lock(db_mutex);
70
71 sqlite3_stmt* stmt = nullptr;
72 sqlite3_prepare_v2(g_db,
73 "INSERT INTO flight_sessions (plane_id, fuel_sum, current_avg, packet_count, status) "
74 "VALUES (?, 0.0, 0.0, 0, 'active');",
75 -1, &stmt, nullptr);
76 sqlite3_bind_int64(stmt, 1, static_cast<int64_t>(session.plane_id));
77 sqlite3_step(stmt);
78 sqlite3_finalize(stmt);
79
80 session.db_row_id = sqlite3_last_insert_rowid(g_db);
81 }
82
83 std::cout << "[flight] Session started | plane_id: " << plane_id << std::endl;
84 return session;

87void process_telemetry(FlightSession& session, double fuel_remaining) {
88 if (!session.has_baseline) {
89 // First packet establishes the baseline lol - no delta
90 session.prev_fuel = fuel_remaining;
91 session.has_baseline = true;
92 session.packet_count = 1;
93 return;
94 }
95
96 double delta = session.prev_fuel - fuel_remaining;
97 session.fuel_sum += delta;
98 session.packet_count++;
99 session.prev_fuel = fuel_remaining;
100 session.current_avg = session.fuel_sum / (session.packet_count - 1);
101
102 if (session.packet_count % 100 == 0)
103 {
104 std::lock_guard<std::mutex> lock(db_mutex);
105
106 sqlite3_stmt* stmt = nullptr;
107
108 sqlite3_prepare_v2(g_db,
109 "UPDATE flight_sessions SET "
110 "fuel_sum = ?, current_avg = ?, packet_count = ? WHERE id = ?;",
111 -1, &stmt, nullptr);
112
113 sqlite3_bind_double(stmt, 1, session.fuel_sum);
114 sqlite3_bind_double(stmt, 2, session.current_avg);
115 sqlite3_bind_int(stmt, 3, session.packet_count);
116 sqlite3_bind_int64(stmt, 4, session.db_row_id);
117
118 sqlite3_step(stmt);
119 sqlite3_finalize(stmt);
120 }

123void finalize_session(FlightSession& session) {
124 if (session.packet_count < 2) {
125 std::cout << "[flight] Session ended | plane_id: " << session.plane_id
126 << " | not enough data for average"
127 << std::endl;
128 return;
129 }
130
131 double avg_consumption = session.fuel_sum / (session.packet_count - 1);
132 double final_cumulative_avg;
133 int final_flight_count;
134
135 std::cout << "[flight] Completed | plane_id: " << session.plane_id
136 << " | avg consumption: " << avg_consumption
137 << " | total consumed: " << session.fuel_sum
138 << " | readings: " << session.packet_count
139 << std::endl;
140
141 {
142 std::lock_guard<std::mutex> lock(aircraft_mutex);
143 AircraftStats& stats = aircraft_map[session.plane_id];
144 stats.cumulative_avg = (stats.cumulative_avg * stats.flight_count + avg_consumption) / (stats.flight_count + 1);
145 stats.flight_count++;
146 final_cumulative_avg = stats.cumulative_avg;
147 final_flight_count = stats.flight_count;
148
149 std::cout << "[fleet] Aircraft "
150 << session.plane_id
151 << " | cumulative avg: " << stats.cumulative_avg
152 << " | flights: " << stats.flight_count
153 << std::endl;
154 }
